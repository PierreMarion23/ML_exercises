Based on the following code, implement a linear discriminant classifier, taking as parameters an $n \times 2$ Numpy array as data and a size-$n$ array of labels.

In [6]:
from sklearn.base import BaseEstimator
from sklearn.discriminant_analysis import LinearClassifierMixin

class LDA(BaseEstimator, LinearClassifierMixin):
    """
        LDA classifier for two classes.
    """
    def __init__(self, prior=None):
        """
            prior: wether to use prior in the intercept. Default is false.
        """
        self.prior = prior
        
    def fit(self, X, y):
        X1 = X[np.where(y==1)]
        X2 = X[np.where(y==-1)]
        mu_pos = np.mean(X1, axis=0)
        mu_neg = np.mean(X2, axis=0)
        self.means_ = np.array([mu_pos, mu_neg])
        pi = self.prior if self.prior else 0.5
        cov = pi*np.cov(X1.T) + (1-pi)*np.cov(X2.T)
        cov_inv = np.linalg.inv(cov)
        self.w = np.dot(cov_inv, mu_pos-mu_neg)
        self.b = np.log(pi / (1-pi)) + 0.5*(np.dot(mu_neg.T, np.dot(cov_inv, mu_neg)) - np.dot(mu_pos.T, np.dot(cov_inv, mu_pos)))

    def decision_function(self, X):
        return np.dot(X, self.w) + self.b

    def predict(self, X):
        return 2*(self.decision_function(X) > 0) - 1
    
    def predict_proba(self, X):
        return np.array(list(map(lambda x: [1 / (1 + np.exp(x)), np.exp(x) / (1 + np.exp(x))], lda.decision_function(X))))
    
    def regret(self, X, y):
        return np.sum(np.abs(self.predict(X) - y)) / len(X)

def plot_decision_clf(clf, X, y, plot_bisector=False):
    plt.figure(figsize=(6, 5))
    plt.scatter(x=X[np.where(y==1)][:,0], y=X[np.where(y==1)][:,1], c='b')
    plt.scatter(x=X[np.where(y==-1)][:,0], y=X[np.where(y==-1)][:,1], c='r')

    nx, ny = 200, 100
    xy_min = np.min(X)
    xy_max = np.max(X)
    plt.xlim([xy_min, xy_max])
    plt.ylim([xy_min, xy_max])
    xx, yy = np.meshgrid(np.linspace(xy_min, xy_max, nx),
                         np.linspace(xy_min, xy_max, ny))
    Z = clf.predict_proba(np.c_[xx.ravel(), yy.ravel()])
    Z = Z[:, 1].reshape(xx.shape)
    plt.pcolormesh(xx, yy, Z, cmap='red_blue_classes',
                   norm=colors.Normalize(0., 1.), zorder=0)
    plt.colorbar()
    plt.contour(xx, yy, Z, [0.5], linewidths=2., colors='white')

    if hasattr(clf, "means_"):
        plt.plot(clf.means_[0][0], clf.means_[0][1], '*', color='white', markersize=15, markeredgecolor='grey')
        plt.plot(clf.means_[1][0], clf.means_[1][1], '*', color='white', markersize=15, markeredgecolor='grey')

        if plot_bisector:
            mu1_exp = clf.means_[0]
            mu2_exp = clf.means_[1]
            x1 = [(mu1_exp[1] - mu2_exp[1])*t + 0.5*(mu2_exp[0] + mu1_exp[0]) for t in range(-100, 200, 100)]
            x2 = [(mu2_exp[0] - mu1_exp[0])*t + 0.5*(mu2_exp[1] + mu1_exp[1]) for t in range(-100, 200, 100)]
            plt.plot(x1, x2, label='bisector of the class centers')
            plt.legend()
    
    plt.show()

Implement the Fisher discriminant analysis based on the following code. In practice, what is the difference between LDA and FisherDA?

class FisherDA(BaseEstimator, LinearClassifierMixin):
    """
        Fisher discriminant analysis for two classes.
    """
    def fit(self, X, y):
        X1 = X[np.where(y==1)]
        X2 = X[np.where(y==-1)]
        mu_pos = np.mean(X1, axis=0)
        mu_neg = np.mean(X2, axis=0)
        self.means_ = np.array([mu_pos, mu_neg])
        pi = len(X1) / len(X)
        cov = pi*np.cov(X1.T) + (1-pi)*np.cov(X2.T)
        cov_inv = np.linalg.inv(cov)
        self.w = np.dot(cov_inv, mu_pos-mu_neg)
        
        # find the optimal b
        proj = np.dot(X, lda.w)
        np.array(list(map(lambda x: 1 if x else -1, proj > 0)))
        self.b = - proj[np.argmin([np.sum(np.abs(2*(proj - b > 0)-1 - y)) for b in proj])]
        
    def decision_function(self, X):
        return np.dot(X, self.w) + self.b

    def predict(self, X):
        return 2*(self.decision_function(X) > 0) - 1
    
    def predict_proba(self, X):
        return np.array(list(map(lambda x: [1 / (1 + np.exp(x)), np.exp(x) / (1 + np.exp(x))], lda.decision_function(X))))
    
    def regret(self, X, y):
        return np.sum(np.abs(self.predict(X) - y)) / len(X)

En pratique, la diffrence entre la LDA et la FDA est qu'on doit choisir b dans le cas de la FDA. Ici on prend b optimal donc on obtient un regret infrieur pour la FDA.

from matplotlib import cm

def map_regions(clf, data=None, num=50):
    """
        Map the regions f(x)=1K of the classifier clf within the same range as the one
        of the data.
        Input:
            clf: classifier with a method predict
            data: input data (X)
            num: discretization parameter
    """
    xmin, ymin = data.min(axis=0)
    xmax, ymax = data.max(axis=0)
    x, y = np.meshgrid(np.linspace(xmin, xmax, num), np.linspace(ymin, ymax))
    z = clf.predict(np.c_[x.ravel(), y.ravel()]).reshape(x.shape)
    zmin, zmax = z.min(), z.max()
    plt.imshow(z, origin='lower', interpolation="nearest",
               extent=[xmin, xmax, ymin, ymax], cmap=cm.coolwarm,
              alpha=0.3)
